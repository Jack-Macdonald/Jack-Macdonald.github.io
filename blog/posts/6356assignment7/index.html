Chart Nine - Scatter Plot

Show the Code 
library(shiny)
library(ggplot2)
library(gapminder)
library(dplyr)
library(RColorBrewer)

data("mtcars")

ui <- fluidPage(
  titlePanel("mtcars Scatter Plot"),
  sidebarLayout(
    sidebarPanel(
      selectInput(inputId = "VarX",
                  label = "Select X-axis Variable:",
                  choices = list("mpg","disp", "hp","drat", "wt","qsec","carb")),
      selectInput(inputId = "VarY",
                  label = "Select Y-axis Variable:",
                  choices = list("disp","mpg", "hp","drat", "wt","qsec","carb")), 
      selectInput(inputId = "Color",
                  label = "Select Color Variable:",
                  choices = as.list(c("am","vs", "gear","cyl")))
    ),
    mainPanel(plotOutput("scatter")
              
    )
  )
)

 server <- shinyServer(function(input, output) {
   
  output$scatter <- renderPlot({
    mtc <- mtcars[ ,c(input$VarX,input$VarY,input$Color)]
    mtc[,3] <- as.factor(mtc[,3])
    
    ggplot(data = mtc, aes(x=mtc[,1],y=mtc[,2],color=mtc[,3],size=4))+
      geom_point()+
      scale_color_manual(values = palette.colors(7,"Dark2"))+
      labs(x=colnames(mtc)[1], y=colnames(mtc)[2],
           color=colnames(mtc)[3],
           title = paste("Scatter Plot of", input$VarX, "v.", input$VarY))+
      guides(size = F)
    
  })
  
})

shinyApp(ui, server)

Chart Eighteen - Table Embedded with Charts

library(shiny)
library(ggplot2)
library(gapminder)
library(dplyr)
library(RColorBrewer)

data("gapminder")

ui <- fluidPage(
  titlePanel("Gapminder Bubble Plot"),
  sidebarLayout(
    sidebarPanel(
      checkboxGroupInput("region", "Select Continent:", unique(gapminder$continent), selected = "Asia"),
      sliderInput("year", "Select Year:", 
                  min = min(gapminder$year), 
                  max = max(gapminder$year), 
                  value = 1952,
                  step = 5,
                  animate = 
                    animationOptions(800,T))
    ),
    mainPanel(
      plotOutput("scatterplot")
    )
  )
)



server <- function(input, output) {
  filtered <- reactive({
    gapminder %>%
      filter(continent %in% input$region, year == input$year)
  })

    output$scatterplot <- renderPlot({
      ggplot(filtered(), aes(gdpPercap, lifeExp, size = pop, col = continent)) +
        geom_point(pch=1, aes(stroke = 1)) +
        scale_size_continuous(range = c(6, 18))+
        scale_color_manual(values = palette.colors(7,"Dark2"))+
        scale_x_continuous(limits=c(0,60000))+
        scale_y_continuous(limits=c(20,90))+
        labs(x = "GDP per Capita", y = "Life Expectancy", 
             title = paste("GPD v. Life Expectancy for Year: ", input$year))+
        theme_bw()
    })
}

shinyApp(ui, server)

Chart Ten - Horizontal Bar Chart

library(shiny)
library(plotly)
library(ggplot2)
library(gapminder)
library(dplyr)
library(RColorBrewer)

data("gapminder")
attach(gapminder)

ui <- fluidPage(
  titlePanel("Gapminder 3D Chart"),
  selectInput("region", "Choose Continent", 
              choices = unique(gapminder$continent), selected = "Asia"),
  plotlyOutput("graph", width = "100%", height = "600px")  
)

server <- function(input, output, session){
  filtered <- reactive({
    gapminder %>%
      filter(continent %in% input$region)
  })
  
  output$graph <- renderPlotly({
    plot_ly(filtered(), 
            x = ~gdpPercap, 
            y = ~lifeExp, 
            z = ~year,
            color = ~year,
            text = ~n,
            hoverinfo = "text",
            hovertext = ~country
            ) %>%
      add_markers()
  })
}

shinyApp(ui, server)
